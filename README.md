# Installation

	$ virtualenv blog
	$ cd blog
	$ source bin/activate
	$ pip install acrylamid
	$ git clone git@github.com:thomaspuppe/blog.thomaspuppe.de.git
	$ git cd blog.thomaspuppe.de
	$ npm install

# Run it

    // during development open two consoles:
    $ acrylamid autocompile
    $ npm run build-css
